// Bubble Sort in IB Pseudocode

N = ARRAY.length
for I from 0 to N-1 do
    for J from 0 to N-I-1 do
        if ARRAY[J] > ARRAY[J+1] then
            // Swap elements at positions J and J+1
            temp = ARRAY[J]
            ARRAY[J] = ARRAY[J+1]
            ARRAY[J+1] = temp
        endif
    endfor
endfor
